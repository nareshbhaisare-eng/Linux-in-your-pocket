#### compile the Linux kernel or its images for specific architectures using a cross-compiler and multiple CPU cores####

#### build device tree blobs (DTBs) for ARM64 platforms, including Raspberry Pi models, using a cross-compiler####
make ARCH=arm64 CROSS_COMPILE=aarch64-linux-gnudtbs -j$(nproc)
make ARCH=arm64 CROSS_COMPILE=aarch64-linux-gnubcm2711- rpi-4-b.dtb

#### build kernel modules and install them to a specified directory using a cross-compiler ####

make ARCH=arm64 CROSS_COMPILE=aarch64-linux-gnu- INSTALL_MOD_PATH=out modules_install

#### nstall compiled kernel modules and the kernel itself on the system ####

sudo make install
